```sql
-- Atomic wallet mutation function (NEVER update wallets directly outside this)
CREATE OR REPLACE FUNCTION public.wallet_mutate(
  p_user_id UUID, p_type TEXT, p_amount NUMERIC, p_ref_type TEXT DEFAULT NULL, p_ref_id UUID DEFAULT NULL, p_description TEXT DEFAULT NULL
) RETURNS public.wallet_transactions AS $$
DECLARE
  v_wallet public.wallets;
  v_new_balance NUMERIC;
  v_txn public.wallet_transactions;
BEGIN
  SELECT * INTO v_wallet FROM public.wallets WHERE user_id = p_user_id FOR UPDATE;
  IF NOT FOUND THEN RAISE EXCEPTION 'Wallet not found for user %', p_user_id; END IF;
  IF v_wallet.is_frozen THEN RAISE EXCEPTION 'Wallet is frozen'; END IF;
  IF p_type IN ('debit','hold','fee') THEN
    IF v_wallet.balance < p_amount THEN RAISE EXCEPTION 'Insufficient balance'; END IF;
    v_new_balance := v_wallet.balance - p_amount;
  ELSE
    v_new_balance := v_wallet.balance + p_amount;
  END IF;
  UPDATE public.wallets SET balance = v_new_balance, updated_at = NOW() WHERE id = v_wallet.id;
  INSERT INTO public.wallet_transactions(wallet_id, user_id, type, amount, balance_before, balance_after, reference_type, reference_id, description)
  VALUES (v_wallet.id, p_user_id, p_type, p_amount, v_wallet.balance, v_new_balance, p_ref_type, p_ref_id, p_description)
  RETURNING * INTO v_txn;
  RETURN v_txn;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

**Migration 005 — trust scores (materialized, computed)**
```sql
CREATE TABLE public.provider_trust_scores (
  provider_id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  score NUMERIC(5,2) NOT NULL DEFAULT 0,
  tier TEXT NOT NULL DEFAULT 'new' CHECK (tier IN ('champion','trusted','verified','rising','new')),
  completion_rate NUMERIC(5,2),
  avg_rating NUMERIC(3,2),
  total_completed INT DEFAULT 0,
  dispute_free_streak INT DEFAULT 0,
  response_time_hours NUMERIC(6,2),
  vouches_count INT DEFAULT 0,
  last_computed_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE public.provider_trust_scores ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public read trust scores" ON public.provider_trust_scores FOR SELECT USING (TRUE);

-- Recompute trust score function (called by trigger on booking completion + review insert)
CREATE OR REPLACE FUNCTION public.recompute_trust_score(p_provider_id UUID) RETURNS VOID AS $$
DECLARE
  v_completed INT; v_total INT; v_avg_rating NUMERIC; v_disputes INT;
  v_score NUMERIC; v_tier TEXT; v_completion_rate NUMERIC;
BEGIN
  SELECT COUNT(*) FILTER (WHERE status = 'completed'),
         COUNT(*),
         AVG(CASE WHEN status = 'completed' THEN 1.0 ELSE 0 END)
  INTO v_completed, v_total, v_completion_rate
  FROM public.bookings WHERE provider_id = p_provider_id;

  SELECT COALESCE(AVG(rating), 0) INTO v_avg_rating FROM public.reviews WHERE provider_id = p_provider_id;
  SELECT COUNT(*) INTO v_disputes FROM public.disputes WHERE against_id = p_provider_id AND status = 'resolved';

  v_score := LEAST(100,
    (v_completion_rate * 40) +
    ((v_avg_rating / 5.0) * 25) +
    (LEAST(v_completed, 100) / 100.0 * 15) +
    (CASE WHEN EXISTS(SELECT 1 FROM public.profiles WHERE id = p_provider_id AND avatar_url IS NOT NULL) THEN 10 ELSE 0 END) +
    (LEAST((SELECT COUNT(*) FROM public.provider_vouches WHERE provider_id = p_provider_id), 5) * 2) -
    (v_disputes * 5)
  );
  v_tier := CASE
    WHEN v_score >= 90 THEN 'champion'
    WHEN v_score >= 78 THEN 'trusted'
    WHEN v_score >= 62 THEN 'verified'
    WHEN v_score >= 45 THEN 'rising'
    ELSE 'new'
  END;
  INSERT INTO public.provider_trust_scores(provider_id, score, tier, completion_rate, avg_rating, total_completed)
  VALUES (p_provider_id, v_score, v_tier, v_completion_rate, v_avg_rating, v_completed)
  ON CONFLICT (provider_id) DO UPDATE SET
    score = EXCLUDED.score, tier = EXCLUDED.tier, completion_rate = EXCLUDED.completion_rate,
    avg_rating = EXCLUDED.avg_rating, total_completed = EXCLUDED.total_completed,
    last_computed_at = NOW();
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

```sql
-- PostGIS proximity search function
CREATE OR REPLACE FUNCTION public.providers_near(p_lat FLOAT, p_lng FLOAT, p_radius_km FLOAT DEFAULT 10)
RETURNS SETOF public.providers AS $$
  SELECT p.* FROM public.providers p
  WHERE ST_DWithin(
    ST_MakePoint(p.longitude, p.latitude)::geography,
    ST_MakePoint(p_lng, p_lat)::geography,
    p_radius_km * 1000
  ) AND p.status = 'approved'
  ORDER BY ST_Distance(
    ST_MakePoint(p.longitude, p.latitude)::geography,
    ST_MakePoint(p_lng, p_lat)::geography
  );
$$ LANGUAGE sql STABLE;
```

### 2.1 `simon-core`
Replaces `src/lib/deepseek.js` entirely. All LLM calls go through this function.
- Reads `DEEPSEEK_API_KEY` from Supabase secrets (never from client env)
- Rate limits: 10 calls/minute per user_id, 100/day (store counter in Redis via Upstash or Supabase KV)
- Input: `{ intent, context, user_id, role }` → Output: `{ result, tokens_used, latency_ms }`
- Intents handled: `home_insights`, `booking_analysis`, `zone_health_narrative`, `receipt_story`, `dispute_summary`, `search_parse`
- If LLM call fails or rate limit exceeded → deterministic heuristic fallback, never an error to client

### 2.2 `emergency-match`
- Reads `emergency_requests` row by id
- Calls `providers_near()` PostGIS function with emergency lat/lng + 5km radius
- Filters: `status = 'approved'`, category match, `is_online = true` (new boolean field on providers)
- Ranks by: distance (40%) + trust_score (35%) + response_time_hours (25%)
- Sends Supabase Realtime broadcast to matched provider's personal channel
- Sends notification row insert for both customer and provider
- Updates `emergency_requests.matched_provider_id` and `status = 'matched'`

### 2.3 `booking-complete`
Triggered by Postgres webhook on `bookings.status` → `completed`:
- Calls `recompute_trust_score(provider_id)` Postgres function
- Calls `award_loyalty_coins()` trigger (already in DB, verify it fires)
- If booking has `recurring_id` → creates next booking instance
- Calls `simon-core` with `receipt_story` intent → writes to `receipt_stories`
- Sends completion notifications to both parties
- Writes to `audit_log`

### 2.4 `wallet-payout`
Called from provider earnings page:
- Validates provider trust tier ≥ 'verified' (no payouts for 'new'/'rising' — fraud prevention)
- Minimum payout: 500 PKR
- Calls `wallet_mutate(provider_id, 'payout', amount)` Postgres function
- Writes `audit_log` entry
- (Payment gateway integration: stub the actual bank transfer call with a structured response for now — mark payout as `pending_transfer` until gateway is integrated)

### 2.5 `zone-health-compute`
Scheduled via pg_cron every 15 minutes:
- For each zone: counts active providers (last seen < 30 mins ago), open bookings, completed bookings today, emergency requests open
- Computes health_score: 0–100
- Updates `neighborhood_zones.health_score` + `demand_index`
- Broadcasts to `zone:{zone_id}` Realtime channel

### 2.6 `trust-passport-generate`
Public endpoint (no auth required):
- Takes `provider_id`
- Returns: full trust credential JSON (score, tier, badges, completion rate, zone map data, dispute-free streak, vouches count)
- Also generates a signed verification hash so third parties can verify the credential hasn't been tampered with: `HMAC-SHA256(credential_json, TRUST_PASSPORT_SECRET)`
- Cached in Supabase KV for 1 hour per provider_id

```javascript
// Real implementation pattern for each hook:
export function useRealtimeTable(table, onChange, filter = null) {
  const [connected, setConnected] = useState(false);
  useEffect(() => {
    let channel = supabase.channel(`realtime:${table}:${filter ?? 'all'}`);
    channel = channel.on('postgres_changes',
      { event: '*', schema: 'public', table, ...(filter ? { filter } : {}) },
      (payload) => onChange(payload)
    );
    channel.subscribe((status) => setConnected(status === 'SUBSCRIBED'));
    return () => { supabase.removeChannel(channel); };
  }, [table, filter]);
  return { connected };
}
```

```javascript
  await supabase.from('profiles').update({ zone_id: zone.id }).eq('id', user.id);
}
```

```sql
-- Postgres function for zone detection:
CREATE OR REPLACE FUNCTION public.find_zone_for_point(p_lat FLOAT, p_lng FLOAT)
RETURNS public.neighborhood_zones AS $$
  SELECT * FROM public.neighborhood_zones
  WHERE ST_Contains(boundary::geometry, ST_MakePoint(p_lng, p_lat)::geometry)
  LIMIT 1;
$$ LANGUAGE sql STABLE;
```

## PHASE 10 — SECURITY HARDENING

### 10.1 Remove `supabaseClient.js` stub fallback in production
The silent stub client is dangerous in production — it returns empty arrays instead of throwing, making misconfiguration invisible. Change to:
```javascript
if (!supabaseUrl || !supabaseAnonKey || supabaseUrl.includes('placeholder')) {
  if (import.meta.env.PROD) throw new Error('FATAL: Supabase credentials not configured');
  console.warn('[DEV] Using stub Supabase client');
  export const supabase = stubClient;
} else {
  export const supabase = createClient(supabaseUrl, supabaseAnonKey, {
    auth: { persistSession: true, autoRefreshToken: true }
  });
}
```

### 10.2 HTTP security headers
Add to `server/index.js` (or Vercel/Netlify config if deploying there):
```javascript
app.use((req, res, next) => {
  res.setHeader('X-Content-Type-Options', 'nosniff');
  res.setHeader('X-Frame-Options', 'DENY');
  res.setHeader('X-XSS-Protection', '1; mode=block');
  res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
  res.setHeader('Permissions-Policy', 'geolocation=(self), camera=(), microphone=(self)');
  next();
});
```

### 10.3 Rate limiting on all server routes
```javascript
import rateLimit from 'express-rate-limit';
const authLimiter = rateLimit({ windowMs: 15 * 60 * 1000, max: 10 }); // 10 login attempts per 15 min
const apiLimiter = rateLimit({ windowMs: 60 * 1000, max: 60 }); // 60 req/min general
app.use('/api/auth', authLimiter);
app.use('/api', apiLimiter);
```

### 10.4 JWT verification on all server routes
```javascript
// middleware/auth.js
import { createClient } from '@supabase/supabase-js';
export async function requireAuth(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  if (!token) return res.status(401).json({ error: 'Unauthorized' });
  const { data: { user }, error } = await supabase.auth.getUser(token);
  if (error || !user) return res.status(401).json({ error: 'Invalid token' });
  req.user = user;
  next();
}
```

Apply `requireAuth` middleware to every `/api/*` route except `/api/trust-passport/:id` (public).
